Algoritmo ContarNumeros
    Definir cantidad, numero, mayores Como Entero
    Definir menores, iguales Como Entero
    
    mayores <- 0
    menores <- 0
    iguales <- 0
	
    Escribir "Introduce la cantidad de números a ingresar:"
    Leer cantidad
	
    Para i Desde 1 Hasta cantidad Hacer
        Escribir "Introduce el número ", i, ":"
        Leer numero
        
        Si numero > 0 Entonces
            mayores <- mayores + 1
        Sino
            Si numero < 0 Entonces
                menores <- menores + 1
            Sino
                iguales <- iguales + 1
            FinSi
        FinSi
    Fin Para
	
    Escribir "Números mayores que 0: ", mayores
    Escribir "Números menores que 0: ", menores
    Escribir "Números iguales a 0: ", iguales
Fin Algoritmo
